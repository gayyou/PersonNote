9. 实现`let a = sum(1).(2, 3).(4)`,`console.log(a.getval())`

   ```js
   function sum(...args) {
     let val = 0;
     for (let item of args) {
       val += item;
     }
     
     inner.getVal = function () {
       return val;
     }
     
     function inner(...args) {
       for (let item of args) {
         val += item;
       }
     }
     
     return inner;
   }
   ```
